14  Turns on and off a light emitting diode(LED) connected to digital

37 // constants won't change. They're used here to
38 // set pin numbers:
39 const int buttonPin = 9; // the number of the pushbutton pin
40 const int ledPin = 13; // the number of the LED pin
41 
42 // variables will change:
43 int buttonState = 0; // variable for reading the pushbutton status
44 
45 void setup() {
46  // initialize the LED pin as an output:
47  pinMode(ledPin, OUTPUT);
48  // initialize the pushbutton pin as an input:
49  pinMode(buttonPin, INPUT);
50 }
51 
52 void loop(){
53  // read the state of the pushbutton value:
54  buttonState = digitalRead(buttonPin);
55 
56  // check if the pushbutton is pressed.
57  // if it is, the buttonState is HIGH:
58  if (buttonState == HIGH) {
59  // turn LED on:
60  digitalWrite(ledPin, HIGH);
61  }
62  else {
63  // turn LED off:
64  digitalWrite(ledPin, LOW);
65  }
66 }
