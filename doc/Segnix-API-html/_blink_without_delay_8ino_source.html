12 /* Blink without Delay
13 
14  Turns on and off a light emitting diode(LED) connected to a digital
15  pin, without using the delay() function. This means that other code
16  can run at the same time without being interrupted by the LED code.
17 
18  The circuit:
19  * LED attached from pin 13 to ground.
20  * Note: on most Arduinos, there is already an LED on the board
21  that's attached to pin 13, so no hardware is needed for this example.

35 // constants won't change. Used here to

52 void loop()
53 {
54  // here is where you'd put code that needs to be running all the time.
55 
56  // check to see if it's time to blink the LED; that is, if the
57  // difference between the current time and last time you blinked
58  // the LED is bigger than the interval at which you want to
59  // blink the LED.
60  unsigned long currentMillis = millis();
61 
62  if(currentMillis - previousMillis > interval) {
63  // save the last time you blinked the LED
64  previousMillis = currentMillis;
65 
66  // if the LED is off turn it on and vice-versa:
67  if (ledState == LOW)
68  ledState = HIGH;
69  else
70  ledState = LOW;
71 
72  // set the LED with the ledState of the variable:
73  digitalWrite(ledPin, ledState);
74  }

76 }
