12 /*
13  Switch statement with serial input
14 
15  Demonstrates the use of a switch statement. The switch
16  statement allows you to choose from among a set of discrete values
17  of a variable. It's like a series of if statements.
18 
19  To see this sketch in action, open the Serial monitor and send any character.
20  The characters a, b, c, d, and e, will turn on LEDs. Any other character will turn
21  the LEDs off.
22 
23  The circuit:
24  * 5 LEDs attached to digital pins 2 through 6 through 220-ohm resistors
25 
26  created 1 Jul 2009
27  by Tom Igoe

29 This example code is in the public domain.
30 
31  http://www.arduino.cc/en/Tutorial/SwitchCase2
32  */

43 void loop() {
44  // read the sensor:
45  if (Serial.available() > 0) {
46  int inByte = Serial.read();
47  // do something different depending on the character received.
48  // The switch statement expects single number values for each case;
49  // in this exmaple, though, you're using single quotes to tell
50  // the controller to get the ASCII value for the character. For
51  // example 'a' = 97, 'b' = 98, and so forth:
52 
53  switch (inByte) {
54  case 'a':
55  digitalWrite(2, HIGH);
56  break;
57  case 'b':
58  digitalWrite(3, HIGH);
59  break;
60  case 'c':
61  digitalWrite(4, HIGH);
62  break;
63  case 'd':
64  digitalWrite(5, HIGH);
65  break;
66  case 'e':
67  digitalWrite(6, HIGH);
68  break;
69  default:
70  // turn all the LEDs off:
71  for (int thisPin = 2; thisPin < 7; thisPin++) {
72  digitalWrite(thisPin, LOW);
73  }
74  }
75  }
76 }
